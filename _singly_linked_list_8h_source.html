135 template <class T>
136 typename SinglyLinkedList<T>::iterator SinglyLinkedList<T> :: insert(const_iterator It, const T& value) {
137  iterator result = iterator{nullptr};
138 
139  /* Time complexity: O(1) */
140  if (It == cbegin()) {
141  addFirst(value);
142  return begin();
143  }
144 
145  /* Time complexity: O(1) */
146  if (It == const_iterator{tail_}) {
147  addLast(value);
148  return iterator{tail_};
149  }
150 
151  /* Time complexity: O(n) */
152  Node<T> *tmp = head_;
153  for (auto cIt = cbegin(); cIt != cend(); ++cIt) {
154  if (It == cIt) {
155  /*Insert new node after the found iterator*/
156  Node<T> *newNode = new Node<T>(value, tmp->next_);
157  tmp->next_ = newNode;
158  size_++;
159  return iterator{newNode};
160  }
161  tmp = tmp->next_;
162  }
163 
164  return result;
165 }
